458  private final BroadcastReceiver changementEtatBluetooth()
459  {
460  BroadcastReceiver receiverEtatBluetooth = new BroadcastReceiver()
461  {
462  @Override
463  public void onReceive(Context context, Intent intent)
464  {
465  final String action = intent.getAction();
466  Log.d(TAG, "[changementEtatBluetooth] action : " + action);
467  if(action.equals(BluetoothAdapter.ACTION_STATE_CHANGED))
468  {
469  final int state = intent.getIntExtra(BluetoothAdapter.EXTRA_STATE, BluetoothAdapter.ERROR);
470  switch(state)
471  {
472  case BluetoothAdapter.STATE_OFF:
473  Log.d(TAG, "[changementEtatBluetooth] bluetooth désactivé !");
474  break;
475  case BluetoothAdapter.STATE_TURNING_OFF:
476  Log.d(TAG, "[changementEtatBluetooth] bluetooth en cours de désactivation !");
477  break;
478  case BluetoothAdapter.STATE_ON:
479  Log.d(TAG, "[changementEtatBluetooth] bluetooth activé !");
480  chercherMinuteur();
481  break;
482  case BluetoothAdapter.STATE_TURNING_ON:
483  Log.d(TAG, "[changementEtatBluetooth] bluetooth en cours d'activation !");
484  break;
485  default:
486  Log.d(TAG, "[changementEtatBluetooth] état : " + state);
487  break;
488  }
489  }
490  else if(action.equals(BluetoothDevice.ACTION_ACL_DISCONNECTED))
491  {
492  Log.d(TAG, "[changementEtatBluetooth] bluetooth déconnecté !");
493  }
494  else if(action.equals(BluetoothDevice.ACTION_BOND_STATE_CHANGED))
495  {
496  final int state = intent.getIntExtra(BluetoothDevice.EXTRA_BOND_STATE, BluetoothDevice.ERROR);
497  switch(state)
498  {
499  case BluetoothDevice.BOND_NONE:
500  Log.d(TAG, "[changementEtatBluetooth] non appairé !");
501  break;
502  case BluetoothDevice.BOND_BONDING:
503  Log.d(TAG, "[changementEtatBluetooth] en cours d'appairage !");
504  break;
505  case BluetoothDevice.BOND_BONDED:
506  Log.d(TAG, "[changementEtatBluetooth] appairé !");
507  break;
508  }
509  }
510  }
511  };
512 
513  return receiverEtatBluetooth;
514  }

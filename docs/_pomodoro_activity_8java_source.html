390  private final BroadcastReceiver changementEtatBluetooth()
391  {
392  BroadcastReceiver receiverEtatBluetooth = new BroadcastReceiver()
393  {
394  @Override
395  public void onReceive(Context context, Intent intent)
396  {
397  final String action = intent.getAction();
398  Log.d(TAG, "[changementEtatBluetooth] action : " + action);
399  if(action.equals(BluetoothAdapter.ACTION_STATE_CHANGED))
400  {
401  final int state = intent.getIntExtra(BluetoothAdapter.EXTRA_STATE, BluetoothAdapter.ERROR);
402  switch(state)
403  {
404  case BluetoothAdapter.STATE_OFF:
405  Log.d(TAG, "[changementEtatBluetooth] bluetooth désactivé !");
406  break;
407  case BluetoothAdapter.STATE_TURNING_OFF:
408  Log.d(TAG, "[changementEtatBluetooth] bluetooth en cours de désactivation !");
409  break;
410  case BluetoothAdapter.STATE_ON:
411  Log.d(TAG, "[changementEtatBluetooth] bluetooth activé !");
412  chercherMinuteur();
413  break;
414  case BluetoothAdapter.STATE_TURNING_ON:
415  Log.d(TAG, "[changementEtatBluetooth] bluetooth en cours d'activation !");
416  break;
417  default:
418  Log.d(TAG, "[changementEtatBluetooth] etat : " + state);
419  break;
420  }
421  }
422  else if(action.equals(BluetoothDevice.ACTION_ACL_DISCONNECTED))
423  {
424  Log.d(TAG, "[changementEtatBluetooth] bluetooth déconnecté !");
425  }
426  else if(action.equals(BluetoothDevice.ACTION_BOND_STATE_CHANGED))
427  {
428  final int state = intent.getIntExtra(BluetoothDevice.EXTRA_BOND_STATE, BluetoothDevice.ERROR);
429  switch(state)
430  {
431  case BluetoothDevice.BOND_NONE:
432  Log.d(TAG, "[changementEtatBluetooth] non appairé !");
433  break;
434  case BluetoothDevice.BOND_BONDING:
435  Log.d(TAG, "[changementEtatBluetooth] en cours d'appairage !");
436  break;
437  case BluetoothDevice.BOND_BONDED:
438  Log.d(TAG, "[changementEtatBluetooth] appairé !");
439  break;
440  }
441  }
442  }
443  };
444 
445  return receiverEtatBluetooth;
446  }
